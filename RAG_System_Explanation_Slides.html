# Semantic Search Implementation def semantic_search(self, query_text: str, limit: int = 10) -> List[Dict]: # Generate embedding for query query_embedding = self.generate_embedding(query_text) # Perform vector similarity search result = self.conn.execute(""" SELECT id, country_name, year, speech_text, array_cosine_similarity(embedding, ?) as similarity FROM speeches WHERE embedding IS NOT NULL ORDER BY similarity DESC LIMIT ? """, [query_embedding, limit]) return [dict(row) for row in result.fetchall()]
